class cow: x,y,id
class event: time,bound,upper,lower (upper,lower ids)
  bound=the blocking cow must have reached this #steps for anything
  to happen, time=the blocked cow reaches this #steps when blocked

input n
array<cow> e_cows n_cows
array<event> events
array ret

populate e_cows, n_cows

sort e_cows, n_cows

for e_cow in e_cows
	for n_cow in n_cows
		if e_cow collides n_cow
			add an event

sort events

for event in events
	if lower is resolved, skip
	if upper is not resolved or upper is past bound
		resolve lower (set ret[lower] to distance lower has moved) 

output ret
